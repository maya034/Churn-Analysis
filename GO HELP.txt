sEmSelMerklePromoHist="Select individual IDs from promo history."
sEmSelMerkleModelScore="Select model scores (static)."

sEmSelAceMetrics="Run system metrics."

sEmSelAceCancelledPldw="Select cancelled policies from PLDW by account number."
sEmSelAceCancelledUnion="Union all cancelled policies by client ID."
sEmSelAceQuoteChub="Select quotes from CHUB by client ID."
sEmSelAceQuoteDcpa="Select quotes from DCPA by client ID."

################################################################################
_all() {

    line1 "$sEmSelAll"

    echo "Starting _all(): $(date '+%F %T')"
    echo "Running as if it were $PLMA_TODAY."
    echo "Running as if yesterday was $PLMA_YESTERDAY."
    echo "Run time is < 1 minute."

    _select
    _derive
    _suppress
    _score
    _report
    _feed

    snow run system-outcome-update.sql

    echo "Finished _all(): $(date '+%F %T')"

}

################################################################################
_select() {

    line1 "$sEmSelSelect"

    sEmSelP2="${1:-$sEmSelP2}" # Direct call overrides the global.

    echo "Now is $(date)"

    echo "--- ACE ---"

    # System metrics.
    snow run system-system_metrics.sql

    # Email lookup tables.
    snow run select-email_from_chub_by_client_id.sql
    snow run select-email_from_pldw_by_qcn.sql

    # Inforce policy data.
    #snow run select-inforce_policies_from_chub_by_client_id.sql
    snow run select-inforce_policies_from_legacy_dcpa_by_client_id.sql
    snow run select-inforce_policies_from_dcpa_by_client_id.sql
    snow run select-inforce_policies_by_client_id.sql

    # Cancelled policy data.
    snow run select-cancelled_policies_from_chub_by_client_id.sql
    snow run select-cancelled_policies_from_dcpa_by_client_id.sql
    snow run select-cancelled_policies_from_pldw_by_account_number.sql
    snow run select-cancelled_policies_by_client_id.sql

    # Quote data.
    snow run select-quotes_from_chub_by_client_id.sql
    snow run select-quotes_from_dcpa_by_client_id.sql
    snow run select-quotes_from_pldw_legacy_by_email.sql
    snow run select-quotes_from_pldw_prevail_by_email.sql
    snow run select-quotes_from_pldw.sql

    # Put together all tables - full eligible universe.
    snow run select-eligible_universe.sql
    # snow run select-delete-junk-emails.sql

    echo "--- MERKLE ---"

    # Promo history.
    snow merkle select-individual_id_from_promo_hist.sql

    # Model scores.
    snow merkle select-model_scores_static.sql

    # Zip suppression.
    snow merkle select-zip_supp.sql

    # Individual summary.
    snow merkle select-universal_individual_summary_vw.sql

}

_suppress() {

    line1 "$sEmSelSuppress"

    # snow run suppress-bad-addresses.sql # Not delete, mark it.

    snow run suppress-dns.sql

################################################################################
_score() {

    line1 "$sEmSelScore"

    snow run assign-test-cases.sql

}

################################################################################
_report() {

    line1 "$sEmSelReport"

    rm -f data/report.tsv  2>/dev/null
    rm -f data/report.xlsx 2>/dev/null

    snow tsv report.sql data/report.tsv
    excel csv2xls data/report.tsv report-template.xlsx data/report.xlsx
    sp put data/report.xlsx "$PLMA_DATA_DELIVERY/Email Selection Process/Email Selection Report - Quote - Current.xlsx"
    sp put data/report.xlsx "$PLMA_DATA_DELIVERY/Email Selection Process/Email Selection Report - Quote - $(date +'%F').xlsx"

}

################################################################################
_feed() {

    line1 "$sEmSelFeed"

    evalx "snow csv feed.sql feed.csv"
    sed -i 's/"//g' feed.csv             # Per legacy, get rid of quotes.
    sed -i '1s/.*/\L&/' feed.csv         # Per legacy, lower case the first row.
    evalx "mv feed.csv $PLMA_MFT/Email_Selection_Process/feed-$PLMA_TODAY.csv"
    evalx "dir $PLMA_MFT/Email_Selection_Process/*"
    evalx "head $PLMA_MFT/Email_Selection_Process/feed-$PLMA_TODAY.csv"

}

################################################################################
_install() {

    line1 "$sAspectInstall"

}

################################################################################
case ${sAction} in

    all)        _all ;;
    derive)     _derive ;;
    feed)       _feed ;;
    help)       usage "$sUsage" ;;
    install)    _install ;;
    report)     _report ;;
    score)      _score ;;
    select)     _select ;;
    suppress)   _suppress ;;
    *)          usage "$sUsage" ;;

esac
